# 编写一个函数将列表子元素顺序反转


			<ul id="target">
			    <li>1</li>
			    <li>2</li>
			    <li>3</li>
			    <li>4</li>
			</ul>

	<script>
	    var target = document.getElementById('target');
	    var i;
	    var frag = document.createDocumentFragment();
	
	    for (i = target.children.length - 1; i >0; --i) {
	        frag.appendChild(target.children[i]);
	    }
	    target.appendChild(frag);
	</script>
